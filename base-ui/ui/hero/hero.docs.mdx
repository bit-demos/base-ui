import { Hero } from './hero';

- A basic hero component with spacing.
- Any child elements or components can be rendered inside.
- A background color or image can be applied with the background prop

### Hero

```js live
<Hero>
  <h1>Stylish Shoes Online</h1>
</Hero>
```

### Hero with background color

```js live
<Hero background="red">
  <h1>Stylish Shoes Online</h1>
</Hero>
```

### Hero with background image

```js live
<Hero background="url(https://images.unsplash.com/photo-1592860986140-d77ede8b7116?ixid=MnwxMjA3fDB8MHxwaG90by1wYWdlfHx8fGVufDB8fHx8&ixlib=rb-1.2.1&auto=format&fit=crop&w=2100&q=80) no-repeat center center scroll">
```
